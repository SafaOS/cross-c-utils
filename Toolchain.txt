# Prefered CC=clang -target x86_64-unknown-none.
# Please set the sysroot.

set(CMAKE_STAGING_PREFIX "${CMAKE_SYSROOT}/usr")
set(UNIX FALSE)
set(CMAKE_SYSTEM_NAME "SafaOS")

set(SAFAOS TRUE)
set(WIN32 FALSE)

set(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
set(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_PACKAGE BOTH)


execute_process(
    COMMAND "${CMAKE_SYSROOT}/usr/bin/config" --cflags --sysroot "${CMAKE_SYSROOT}"
    OUTPUT_VARIABLE SYSROOT_CFLAGS
    OUTPUT_STRIP_TRAILING_WHITESPACE
)

execute_process(
    COMMAND "${CMAKE_SYSROOT}/usr/bin/config" --ldflags --sysroot "${CMAKE_SYSROOT}"
    OUTPUT_VARIABLE SYSROOT_LDFLAGS
    OUTPUT_STRIP_TRAILING_WHITESPACE
)


set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${SYSROOT_CFLAGS}")

# Linker flags to use for executables and shared libs
set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} ${SYSROOT_LDFLAGS}")
set(CMAKE_SHARED_LINKER_FLAGS "${CMAKE_SHARED_LINKER_FLAGS} ${SYSROOT_LDFLAGS}")
set(CMAKE_SIZEOF_VOID_P 8)
